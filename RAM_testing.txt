// Mouse and System Variables
@0x00   0x00   // Reserved
@0x01   0xA0   // Mouse status byte
@0x02   0x00   // Reserved
@0x03   0x00   // Reserved
@0x04   0x77   // System config
@0x05   0x01   // Constant 1
@0x06   0x00   // Reserved
@0x07   0x00   // Reserved
@0x08   0x00   // Reserved
@0x09   0xFF   // Constant FF

// Grid Thresholds (for 160x120 grid)
@0x10   0x35   // X1 = 53 (Left/Middle)
@0x11   0x6B   // X2 = 107 (Middle/Right)
@0x12   0x28   // Y1 = 40 (Front/Middle)
@0x13   0x50   // Y2 = 80 (Middle/Back)

// Mouse Position Storage
@0x14   0x00   // Current X position
@0x15   0x00   // Current Y position
@0x16   0x00   // Previous pixel value

// Direction Commands (matches 7-seg encodings)
@0x17   0x0A   // BL - L code
@0x18   0x0B   // BR - R code
@0x19   0x0D   // B - B code
@0x20   0x0A   // FL - L code
@0x21   0x0B   // FR - R code
@0x22   0x0C   // F - F code
@0x23   0x0A   // L - L code
@0x24   0x0B   // R - R code
@0x25   0x01   // I - '1' code
